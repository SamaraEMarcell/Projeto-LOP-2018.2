var x=15
var y=470
function setup() {
  createCanvas(500, 500);
  frameRate(30); 
}
function draw() {
  background(200)
  fill(255,0,0);
  stroke(0, 0, 255);
  ellipse(x,y,30,60); 
  if ( keyIsDown(RIGHT_ARROW) ) 
  {
  x = x + 4;
  }
  if( keyIsDown(LEFT_ARROW))
  {
    x = x - 4;
  }
}
